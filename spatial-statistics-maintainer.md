# 获取 R 包元数据
Sys.setenv(R_CRAN_WEB = "https://mirrors.tuna.tsinghua.edu.cn/CRAN")
# 返回 data.frame
pdb <- tools::CRAN_package_db()

gh_repo <- function(x) {
  x <- lapply(strsplit(x, split = ","), trimws)
  unlist(lapply(x, function(x) x[grepl(pattern = "github.com", x)]))
}

# 提取描述中提及 [Ss]patial 的 R 包列表
sub_pdb <- subset(
  x = pdb, subset = grepl(x = Description, "[Ss]patial"),
  select = c("Maintainer", "Package", "Title", "URL", "BugReports")
) |>
  transform(Title = gsub("(\\\n)", " ", Title)) |>
  transform(Maintainer = gsub("<([^<>]*)>", "", Maintainer)) |>
  transform(URL = gsub("(\\\n)", " ", URL)) |>
  transform(URL = ifelse(is.na(URL), sprintf("https://CRAN.R-project.org/package=%s", Package), URL)) |>
  transform(URL = ifelse(grepl(pattern = "github.com", URL) , gh_repo(URL), URL)) |> 
  transform(Package = paste0("[","**", Package, "**", "]", "(", URL,")")) |> 
  subset(select = c("Maintainer", "Package", "Title"))

sub_pdb2 <- sub_pdb[order(sub_pdb$Maintainer), ]

knitr::kable(sub_pdb2,
  row.names = FALSE,
  caption = "空间统计相关的 R 包作者",
  col.names = c("维护者", "R 包", "简介")
)
